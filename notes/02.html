# webpack 中配置别名

resolve: {
  extensions: ['.js', '.vue', '.json'],
  alias: {
    '@': resolve('src'),
  }
},

用法举例

在.vue 文件中使用：

<style scoped lang="stylus" rel="stylesheet/stylus">
  @import "~@/common/stylus/variable"

  #app
    color: $color-theme
</style>

在.js 文件中使用：

import '@/common/stylus/index.styl'
